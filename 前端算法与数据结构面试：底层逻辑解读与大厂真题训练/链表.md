- 处理链表的本质，是处理链表结点之间的指针关系

### 链表合并

- 循环

```javascript
var mergeTwoLists = function (l1, l2) {
  let head = new ListNode(); // // 头节点
  let cur = head; // 指针
  // 遍历两个链表，cur的next指向val小的节点。
  // val小的节点链表节点后移，cur后移
  while (l1 && l2) {
    // ...
  }
  // 连接未遍历部分
  cur.next = l1 ? l1 : l2;
  return head.next;
};
```

- 递归

```javascript
var mergeTwoLists = function (l1, l2) {
  // 一个链表为空时，返回另外一个（递归终止条件）
  if (!l1) return l2;
  if (!l2) return l1;
  // 选择小的节点（递归函数）
  if (l1.val < l2.value) {
    l1.next = mergeTwoLists(l1.next, l2);
    return l1;
  }
  // 同上
};
```
